/** * ============================================================ * كود الباك إند المطور - مؤسسة مصر بكرة للتنمية * الإصدار: 3.0 (تحديث الأداء اليومي والكلمات المفتاحية) * ============================================================ */ function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // تكوين الجداول شاملة Settings و DailyPerformance const sheetsConfig = [ { name: "Users", headers: ["Username", "Password", "Name", "Role", "LeaveBalance", "PhotoURL"], color: "#2563eb" }, { name: "Attendance", headers: ["Timestamp", "StudentIDs", "AttendanceDate", "Programs", "Teachers", "Notes"], color: "#16a34a" }, { name: "LeaveRequests", headers: ["ReqID", "TeacherName", "LeaveDate", "Type", "Status", "SubmissionDate"], color: "#ea580c" }, { name: "Students", headers: ["StudentID", "StudentName", "Shift"], color: "#0284c7" }, { name: "Settings", headers: ["البرنامج", "الكلمة المفتاحية", "العداد"], color: "#8b5cf6" }, { name: "DailyPerformance", headers: ["Date", "Teacher", "Program", "Keywords"], color: "#f43f5e" } ]; sheetsConfig.forEach(sh => { let sheet = ss.getSheetByName(sh.name); if (!sheet) { sheet = ss.insertSheet(sh.name); sheet.getRange(1, 1, 1, sh.headers.length).setValues([sh.headers]) .setBackground(sh.color).setFontColor("white").setFontWeight("bold"); } else { // تحديث الهيدرز للتأكد من مطابقتها للتحديثات sheet.getRange(1, 1, 1, sh.headers.length).setValues([sh.headers]); } sheet.setFrozenRows(1); try { sheet.setRightToLeft(true); } catch(e) {} }); // إضافة بيانات تجريبية لشيت الإعدادات إذا كان فارغاً const settingsSheet = ss.getSheetByName("Settings"); if (settingsSheet.getLastRow() === 1) { const demoData = [ ["تعليم", "مستوى مبتدئ", 0], ["تعليم", "مراجعة شاملة", 0], ["رياضة", "لياقة بدنية", 0], ["نشاط", "تفاعل اجتماعي", 0] ]; demoData.forEach(row => settingsSheet.appendRow(row)); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; switch(action) { case "login": return handleLogin(data.username, data.password); case "getDashboard": return getDashboardData(data.username, data.role, data.name, data.startDate, data.endDate); case "submitAttendance": return submitAttendance(data); case "requestLeave": return requestLeave(data); case "manageLeave": return manageLeave(data); case "addManualLeave": return addManualLeave(data); case "submitDailyPerformance": return submitDailyPerformance(data); default: return respond({status: "error", message: "العملية المطلوبة غير مدعومة"}); } } catch (error) { return respond({status: "error", message: "خطأ: " + error.toString()}); } } function respond(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function handleLogin(username, password) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Users"); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === username.toString() && data[i][1].toString() === password.toString()) { return respond({ status: "success", user: { username: data[i][0], name: data[i][2], role: data[i][3], leaveBalance: data[i][4], photo: data[i][5] } }); } } return respond({status: "error", message: "بيانات الدخول غير صحيحة"}); } function submitAttendance(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Attendance"); let rawIds = data.ids.toString().replace(/[\n\r\s,]+/g, ' ').trim(); let formattedIds = rawIds.split(' ').filter(id => id.length > 0).join(' . '); sheet.appendRow([ new Date().toISOString(), formattedIds, data.customDate || Utilities.formatDate(new Date(), "GMT+2", "yyyy-MM-dd"), data.programs, data.teachers, data.notes || "" ]); return respond({status: "success", message: "تم تسجيل الحضور بنجاح"}); } function submitDailyPerformance(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const perfSheet = ss.getSheetByName("DailyPerformance"); const settingsSheet = ss.getSheetByName("Settings"); // 1. تسجيل الحدث perfSheet.appendRow([ data.date || Utilities.formatDate(new Date(), "GMT+2", "yyyy-MM-dd"), data.teacherName, data.program, data.keywords.join(" , ") ]); // 2. تحديث العداد (+1) في شيت Settings const sData = settingsSheet.getDataRange().getValues(); for (let k of data.keywords) { let found = false; for (let i = 1; i < sData.length; i++) { if (sData[i][0] === data.program && sData[i][1] === k) { let currentCount = parseInt(sData[i][2]) || 0; settingsSheet.getRange(i + 1, 3).setValue(currentCount + 1); found = true; break; } } // إضافة الكلمة إذا لم تكن موجودة مسبقاً للحماية if (!found) { settingsSheet.appendRow([data.program, k, 1]); } } return respond({status: "success", message: "تم تسجيل الأداء وتحديث المؤشرات بنجاح"}); } function getDashboardData(username, role, name, startDate, endDate) { const ss = SpreadsheetApp.getActiveSpreadsheet(); initSheets(); // تأكد من وجود الشيتات const attendanceData = ss.getSheetByName("Attendance").getDataRange().getValues(); const leaveData = ss.getSheetByName("LeaveRequests").getDataRange().getValues(); const userData = ss.getSheetByName("Users").getDataRange().getValues(); const studentData = ss.getSheetByName("Students").getDataRange().getValues(); const settingsDataRows = ss.getSheetByName("Settings").getDataRange().getValues(); const perfDataRows = ss.getSheetByName("DailyPerformance").getDataRange().getValues(); let now = new Date(); let y = now.getFullYear(); let m = ("0" + (now.getMonth() + 1)).slice(-2); let dLast = new Date(y, now.getMonth() + 1, 0).getDate(); let filterStart = startDate || `${y}-${m}-01`; let filterEnd = endDate || `${y}-${m}-${("0" + dLast).slice(-2)}`; let studentMap = {}; let studentShiftMap = {}; for(let i = 1; i < studentData.length; i++) { studentMap[studentData[i][0]] = studentData[i][1]; studentShiftMap[studentData[i][0]] = studentData[i][2]; } // جلب الإعدادات (الكلمات المفتاحية) let settingsList = []; for(let i=1; i= filterStart && dateStr <= filterEnd) { dailyPerformances.push({ date: dateStr, teacher: perfDataRows[i][1], program: perfDataRows[i][2], keywords: perfDataRows[i][3] }); } } let studentStats = {}; let programStats = {}; let centerDays = new Set(); let allUsers = []; let teachersReportObj = {}; const excludedUsers = ["salwa", "amer"]; for(let i = 1; i < userData.length; i++) { let uName = userData[i][2]; let uRole = userData[i][3]; if(excludedUsers.includes(uName.toLowerCase())) continue; allUsers.push({ name: uName, role: uRole }); teachersReportObj[uName] = { presentDays: new Set(), approvedLeaveDates: new Set(), photo: userData[i][5], balance: userData[i][4], programs: {} }; } let pendingLeaves = []; for(let i = 1; i < leaveData.length; i++) { let tName = leaveData[i][1]; let lDate = leaveData[i][2]; let lStatus = leaveData[i][4]; if(lStatus === "Pending") { pendingLeaves.push({id: leaveData[i][0], teacher: tName, date: lDate, type: leaveData[i][3], status: lStatus}); } let dVal = new Date(lDate); if(!isNaN(dVal.getTime())) { let dateStr = Utilities.formatDate(dVal, "GMT+2", "yyyy-MM-dd"); if(dateStr >= filterStart && dateStr <= filterEnd) { if(lStatus === "Approved" && teachersReportObj[tName]) { teachersReportObj[tName].approvedLeaveDates.add(dateStr); } } } } let uniqueLectures = new Set(); for(let i = 1; i < attendanceData.length; i++) { let rawDate = attendanceData[i][2]; if(!rawDate) continue; let dVal = new Date(rawDate); if(isNaN(dVal.getTime())) continue; let dateStr = Utilities.formatDate(dVal, "GMT+2", "yyyy-MM-dd"); if(dateStr < filterStart || dateStr > filterEnd) continue; centerDays.add(dateStr); let progs = attendanceData[i][3].toString().split(' - ').map(s => s.trim()).filter(Boolean); let teachers = attendanceData[i][4].toString().split(' - ').map(s => s.trim()).filter(Boolean); let ids = attendanceData[i][1].toString().split(' . ').filter(id => id.length > 0); teachers.forEach(t => { if(excludedUsers.includes(t.toLowerCase()) || !teachersReportObj[t]) return; teachersReportObj[t].presentDays.add(dateStr); progs.forEach(p => { let lecKey = `${dateStr}_${p}_${t}`; if(!uniqueLectures.has(lecKey)) { uniqueLectures.add(lecKey); teachersReportObj[t].programs[p] = (teachersReportObj[t].programs[p] || 0) + 1; if(!programStats[p]) programStats[p] = { days: new Set(), lectures: 0 }; programStats[p].lectures++; } }); }); progs.forEach(p => { if(p) { if(!programStats[p]) programStats[p] = { days: new Set(), lectures: 0 }; programStats[p].days.add(dateStr); } }); ids.forEach(id => { if(!studentStats[id]) studentStats[id] = { days: new Set(), lectures: 0, programs: new Set() }; studentStats[id].days.add(dateStr); studentStats[id].lectures++; progs.forEach(p => { if(p) studentStats[id].programs.add(p); }); }); } // التعديل 1: معادلة حساب أيام الغياب المحدثة let teachersReportFinal = Object.keys(teachersReportObj).map(tName => { let pDays = teachersReportObj[tName].presentDays; let lDatesSet = teachersReportObj[tName].approvedLeaveDates; let leaveDatesArray = Array.from(lDatesSet).sort(); let pDaysCount = pDays.size; let leaveDaysCount = lDatesSet.size; // المعادلة الجديدة: (أيام العمل الفعلية × 0.75) - أيام الإجازات let calculatedAbsent = (pDaysCount * 0.75) - leaveDaysCount; // الشرط: إذا كان بالسالب، يعرض 0 let finalAbsentDays = Math.max(0, calculatedAbsent); return { name: tName, photo: teachersReportObj[tName].photo, balance: teachersReportObj[tName].balance, presentDays: pDaysCount, leaveDays: leaveDaysCount, leaveDates: leaveDatesArray, absentDays: finalAbsentDays.toFixed(2), // عرض برقمين عشريين للوضوح programs: teachersReportObj[tName].programs }; }); return respond({ status: "success", data: { users: allUsers, totalWorkDays: centerDays.size, students: Object.keys(studentStats).map(id => ({ id: id, name: studentMap[id] || "غير مسجل", shift: studentShiftMap[id] || "غير محدد", days: studentStats[id].days.size, lectures: studentStats[id].lectures, programs: Array.from(studentStats[id].programs) })), programs: Object.keys(programStats).map(p => ({ name: p, days: programStats[p].days.size, lectures: programStats[p].lectures })), teachersReport: teachersReportFinal, myReport: teachersReportFinal.find(t => t.name === name) || null, myLeaves: leaveData.filter(r => r[1] === name).map(r => ({id: r[0], date: r[2], type: r[3], status: r[4]})), pendingLeaves: role === "Admin" ? pendingLeaves : [], pendingLeavesCount: pendingLeaves.length, settings: settingsList, performances: dailyPerformances } }); } function requestLeave(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("LeaveRequests"); const reqId = "REQ-" + new Date().getTime(); sheet.appendRow([ reqId, data.teacherName, data.leaveDate, data.leaveType, "Pending", Utilities.formatDate(new Date(), "GMT+2", "yyyy-MM-dd") ]); return respond({status: "success", message: "تم إرسال الطلب للإدارة بنجاح"}); } function addManualLeave(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("LeaveRequests"); const reqId = "MAN-" + new Date().getTime(); sheet.appendRow([ reqId, data.teacherName, data.leaveDate, data.leaveType, "Approved", Utilities.formatDate(new Date(), "GMT+2", "yyyy-MM-dd") ]); if (data.leaveType === "اعتيادي" || data.leaveType === "عارضة") { deductLeaveBalance(data.teacherName); } return respond({status: "success", message: "تم تسجيل الإجازة بنجاح"}); } function manageLeave(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("LeaveRequests"); const rows = sheet.getDataRange().getValues(); for(let i = 1; i < rows.length; i++) { if(rows[i][0].toString() === data.reqId.toString()) { sheet.getRange(i + 1, 5).setValue(data.newStatus); if(data.newStatus === "Approved") { let type = rows[i][3]; if(type === "اعتيادي" || type === "عارضة") { deductLeaveBalance(rows[i][1]); } } return respond({status: "success", message: "تم التحديث بنجاح"}); } } return respond({status: "error", message: "الطلب غير موجود"}); } function deductLeaveBalance(teacherName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Users"); const data = sheet.getDataRange().getValues(); for(let i = 1; i < data.length; i++) { if(data[i][2] === teacherName) { let currentBalance = parseFloat(data[i][4]) || 0; sheet.getRange(i + 1, 5).setValue(Math.max(0, currentBalance - 1)); break; } } }
